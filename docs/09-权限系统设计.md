# 09 - 权限系统设计文档

## 1. 权限系统概述

MOCHU-OA采用**RBAC（基于角色的访问控制）**权限模型，通过角色关联用户和权限，实现灵活的权限管理。

## 2. RBAC模型

### 2.1 核心概念

```
用户(User) ──┬── 用户角色关联(user_roles) ──── 角色(Role)
             │                                    │
             └────────────────────────────────────┴── 角色权限(permissions)
```

### 2.2 关系图

```
┌─────────────┐
│   users     │
│  (用户表)   │
└──────┬──────┘
       │
       │ 1:N
       │
┌──────▼──────┐      N:1     ┌─────────────┐
│ user_roles  │──────────────▶│   roles     │
│ (用户角色)  │               │  (角色表)   │
└─────────────┘               └──────┬──────┘
                                     │
                                     │ permissions (JSON)
                                     │
                              ┌──────▼──────┐
                              │ permissions │
                              │  (权限列表)  │
                              └─────────────┘
```

## 3. 数据库设计

### 3.1 用户表 (users)

```sql
CREATE TABLE users (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  username TEXT UNIQUE NOT NULL,          -- 用户名（唯一）
  password TEXT NOT NULL,                 -- 密码（bcrypt加密）
  real_name TEXT,                         -- 真实姓名
  phone TEXT,                             -- 手机号
  email TEXT,                             -- 邮箱
  status TEXT DEFAULT 'active',           -- 状态: active, inactive
  department_id INTEGER,                  -- 部门ID
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  last_login_at DATETIME,
  
  FOREIGN KEY (department_id) REFERENCES departments(id)
);
```

### 3.2 角色表 (roles)

```sql
CREATE TABLE roles (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name TEXT NOT NULL,                     -- 角色名称（显示名）
  code TEXT UNIQUE NOT NULL,              -- 角色代码（唯一）
  description TEXT,                       -- 角色描述
  permissions TEXT,                       -- 权限列表（JSON数组）
  status TEXT DEFAULT 'active',           -- 状态
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
```

**permissions字段示例：**
```json
[
  "project:view",
  "project:create",
  "project:edit",
  "contract:view",
  "contract:approve"
]
```

### 3.3 用户角色关联表 (user_roles)

```sql
CREATE TABLE user_roles (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  user_id INTEGER NOT NULL,
  role_id INTEGER NOT NULL,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
  FOREIGN KEY (role_id) REFERENCES roles(id) ON DELETE CASCADE,
  UNIQUE(user_id, role_id)
);
```

## 6. 权限检查实现

### 6.1 获取用户权限

```javascript
/**
 * 获取用户所有权限
 * @param {number} userId - 用户ID
 * @returns {string[]} 权限代码数组
 */
function getUserPermissions(userId) {
  const db = require('better-sqlite3')('./database.db');
  
  // 查询用户的所有角色
  const roles = db.prepare(`
    SELECT r.id, r.code, r.name, r.permissions
    FROM roles r
    INNER JOIN user_roles ur ON r.id = ur.role_id
    WHERE ur.user_id = ?
  `).all(userId);

  // 检查是否是超级管理员(GM)
  const isGM = roles.some(role => role.code === 'GM' || role.code === 'gm');
  if (isGM) {
    // 超级管理员拥有所有权限
    return ['*'];
  }

  // 合并所有角色的权限
  const permissions = new Set();
  
  roles.forEach(role => {
    // 角色代码本身也是一种权限标识
    permissions.add(role.code.toLowerCase());
    
    // 从角色的 permissions JSON 字段中获取具体权限
    if (role.permissions) {
      try {
        const rolePermissions = JSON.parse(role.permissions);
        if (Array.isArray(rolePermissions)) {
          rolePermissions.forEach(perm => {
            permissions.add(perm);
          });
        }
      } catch (e) {
        console.error('解析角色权限失败:', e.message);
      }
    }
  });

  return Array.from(permissions);
}
```

### 6.2 检查单个权限

```javascript
/**
 * 检查用户是否拥有指定权限
 * @param {string[]} userPermissions - 用户权限列表
 * @param {string} requiredPermission - 需要的权限
 * @returns {boolean}
 */
function hasPermission(userPermissions, requiredPermission) {
  // 超级管理员拥有所有权限
  if (userPermissions.includes('*')) {
    return true;
  }
  
  return userPermissions.includes(requiredPermission);
}
```

### 6.3 检查多个权限（满足其一）

```javascript
/**
 * 检查用户是否拥有任意一个指定权限
 * @param {string[]} userPermissions - 用户权限列表
 * @param {string[]} requiredPermissions - 需要的权限列表
 * @returns {boolean}
 */
function hasAnyPermission(userPermissions, requiredPermissions) {
  // 超级管理员拥有所有权限
  if (userPermissions.includes('*')) {
    return true;
  }
  
  return requiredPermissions.some(perm => 
    userPermissions.includes(perm)
  );
}
```

### 6.4 检查所有权限

```javascript
/**
 * 检查用户是否拥有所有指定权限
 * @param {string[]} userPermissions - 用户权限列表
 * @param {string[]} requiredPermissions - 需要的权限列表
 * @returns {boolean}
 */
function hasAllPermissions(userPermissions, requiredPermissions) {
  // 超级管理员拥有所有权限
  if (userPermissions.includes('*')) {
    return true;
  }
  
  return requiredPermissions.every(perm => 
    userPermissions.includes(perm)
  );
}
```

## 7. 权限中间件

### 7.1 单个权限检查中间件

```javascript
/**
 * 权限检查中间件工厂函数
 * @param {string} permission - 需要的权限
 * @returns {Function} Express中间件
 * 
 * @example
 * router.post('/projects', 
 *   authMiddleware, 
 *   checkPermission('project:create'), 
 *   createProject
 * );
 */
function checkPermission(permission) {
  return (req, res, next) => {
    // 检查用户是否登录
    if (!req.user) {
      return res.status(401).json({
        success: false,
        message: '未登录'
      });
    }

    // 获取用户权限
    const userPermissions = getUserPermissions(req.user.id);
    
    // 检查权限
    if (!hasPermission(userPermissions, permission)) {
      console.log(`权限检查失败: 用户 ${req.user.username} 没有 ${permission} 权限`);
      return res.status(403).json({
        success: false,
        message: '无权限执行此操作',
        required: permission
      });
    }

    // 权限检查通过
    req.userPermissions = userPermissions;
    next();
  };
}
```

### 7.2 多个权限检查中间件（满足其一）

```javascript
/**
 * 检查是否拥有任意一个权限
 * @param {string[]} permissions - 权限列表
 * @returns {Function} Express中间件
 */
function checkAnyPermission(permissions) {
  return (req, res, next) => {
    if (!req.user) {
      return res.status(401).json({
        success: false,
        message: '未登录'
      });
    }

    const userPermissions = getUserPermissions(req.user.id);
    
    if (!hasAnyPermission(userPermissions, permissions)) {
      return res.status(403).json({
        success: false,
        message: '无权限执行此操作',
        required: permissions
      });
    }

    req.userPermissions = userPermissions;
    next();
  };
}
```

### 7.3 角色检查中间件

```javascript
/**
 * 检查用户是否拥有指定角色
 * @param {string|string[]} roles - 角色代码或数组
 * @returns {Function} Express中间件
 */
function checkRole(roles) {
  const roleArray = Array.isArray(roles) ? roles : [roles];
  
  return (req, res, next) => {
    if (!req.user) {
      return res.status(401).json({
        success: false,
        message: '未登录'
      });
    }

    // 获取用户角色
    const userRoles = getUserRoleCode(req.user.id);
    
    // 检查角色
    const hasRole = roleArray.some(role => 
      userRoles.includes(role.toUpperCase()) || userRoles.includes(role.toLowerCase())
    );
    
    if (!hasRole) {
      return res.status(403).json({
        success: false,
        message: '无权限执行此操作',
        required: roleArray
      });
    }

    next();
  };
}

/**
 * 获取用户角色代码列表
 * @param {number} userId 
 * @returns {string[]}
 */
function getUserRoleCode(userId) {
  const db = require('better-sqlite3')('./database.db');
  
  const roles = db.prepare(`
    SELECT r.code
    FROM roles r
    INNER JOIN user_roles ur ON r.id = ur.role_id
    WHERE ur.user_id = ?
  `).all(userId);
  
  return roles.map(r => r.code);
}
```

```javascript
// 权限Hook
function usePermission() {
  const { user } = useAuth();
  const [permissions, setPermissions] = useState([]);
  
  useEffect(() => {
    if (user) {
      const perms = getUserPermissions(user.id);
      setPermissions(perms);
    }
  }, [user]);
  
  const hasPerm = useCallback((permission) => {
    return hasPermission(permissions, permission);

  const hasAnyPerm = useCallback((permList) => {
    return hasAnyPermission(permissions, permList);
  }, [permissions]);
  
  return {
    permissions,
    hasPermission: hasPerm,
    hasAnyPermission: hasAnyPerm,
    isAdmin: permissions.includes('*')
  };
}
```

## 10. 权限缓存

### 10.1 内存缓存

```javascript
// 权限缓存
const permissionCache = new Map();
const CACHE_TTL = 5 * 60 * 1000; // 5分钟

function getUserPermissionsWithCache(userId) {
  const cacheKey = `user_perms_${userId}`;
  const cached = permissionCache.get(cacheKey);
  
  if (cached && Date.now() - cached.timestamp < CACHE_TTL) {
    return cached.permissions;
  }
  
  const permissions = getUserPermissions(userId);
  permissionCache.set(cacheKey, {
    permissions,
    timestamp: Date.now()
  });
  
  return permissions;
}

// 清除缓存
function clearPermissionCache(userId) {
  if (userId) {
    permissionCache.delete(`user_perms_${userId}`);
  } else {
    permissionCache.clear();
  }
}
```

### 10.2 缓存更新

```javascript
// 用户角色变更时清除缓存
async function updateUserRoles(userId, roleIds) {
  const db = require('better-sqlite3')('./database.db');
  
  // 删除旧角色
  db.prepare('DELETE FROM user_roles WHERE user_id = ?').run(userId);
  
  // 添加新角色
  const insertStmt = db.prepare(`
    INSERT INTO user_roles (user_id, role_id) VALUES (?, ?)
  `);
  
  for (const roleId of roleIds) {
    insertStmt.run(userId, roleId);
  }
  
  // 清除权限缓存
  clearPermissionCache(userId);
}
```

## 12. 最佳实践

### 12.1 权限设计原则

1. **最小权限原则**: 只授予完成工作所需的最小权限
2. **职责分离**: 敏感操作需要多人审批
3. **权限继承**: 子角色可以继承父角色的权限
4. **权限过期**: 临时权限应设置过期时间

### 12.2 安全建议

1. **定期审计**: 定期检查用户权限是否合理
2. **权限回收**: 员工离职或调岗时及时回收权限
3. **操作日志**: 记录所有权限相关操作
4. **多重验证**: 敏感操作需要二次验证

### 12.3 常见问题

**Q: 如何处理权限冲突？**
A: 权限是累加的，用户拥有所有角色的权限并集。

**Q: 如何实现数据权限？**
A: 在查询时根据用户所属部门/项目过滤数据。

**Q: 如何实现临时授权？**
A: 创建临时角色或设置权限过期时间。
