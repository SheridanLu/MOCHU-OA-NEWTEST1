# 08 - 审批流程设计文档

## 1. 审批流程概述

MOCHU-OA系统采用多级审批流程设计，支持21种不同的审批类型，每种类型可配置不同的审批节点和流程。

## 3. 审批流程设计

### 3.1 两级审批流程（最常见）

```
┌──────────┐    提交     ┌──────────┐   通过    ┌──────────┐   通过   ┌──────────┐
│  申请人  │ ─────────→ │  财务管理 │ ───────→ │  总经理  │ ───────→ │  完成    │
└──────────┘            └──────────┘          └──────────┘         └──────────┘
                              │                     │
                              │ 拒绝                │ 拒绝
                              ↓                     ↓
                        ┌──────────┐         ┌──────────┐
                        │  已拒绝  │         │  已拒绝  │
                        └──────────┘         └──────────┘
```

**适用审批类型：**
- 项目立项
- 虚拟转实体
- 虚拟中止
- 零星采购
- 采购清单
- 批量采购

- 劳务付款
- 材料变更
- 签证变更
- 业主变更
- 竣工结算
- 超量申请
- 劳务签证
- 入库单
- 零星采购（新）
- 现场签证
- 超量审批

## 4. 审批状态定义

| 状态 | 代码 | 说明 | 可执行操作 |
|------|------|------|-----------|
| 草稿 | draft | 未提交审批 | 编辑、删除、提交 |
| 审批中 | pending | 等待审批 | 查看、审批、拒绝 |
| 已通过 | approved | 审批完成 | 查看 |
| 已拒绝 | rejected | 审批被拒绝 | 查看、重新提交 |
| 已完成 | completed | 业务完成 | 查看 |
| 已取消 | cancelled | 已取消 | 查看 |

## 5. 数据库设计

### 5.1 通用审批表结构

系统为不同审批类型创建了独立的审批表，但结构基本一致：

```sql
CREATE TABLE xxx_approvals (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  
  -- 关联信息
  xxx_id INTEGER NOT NULL,           -- 关联的业务ID（项目/合同/清单等）
  project_id INTEGER,                 -- 关联项目ID
  
  -- 审批信息
  approval_no TEXT UNIQUE,            -- 审批编号
  approval_step INTEGER DEFAULT 0,    -- 当前审批步骤
  current_approver TEXT,              -- 当前审批人角色代码
  current_approver_id INTEGER,        -- 当前审批人用户ID
  
  -- 状态信息
  status TEXT DEFAULT 'pending',      -- 审批状态
  submitter_id INTEGER,               -- 提交人ID
  submitted_at DATETIME,              -- 提交时间
  
  -- 审批意见
  approval_comment TEXT,              -- 审批意见
  rejected_reason TEXT,               -- 拒绝原因
  
  -- 时间戳
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  
  FOREIGN KEY (xxx_id) REFERENCES xxx_table(id),
  FOREIGN KEY (project_id) REFERENCES projects(id)
);
```

### 5.2 审批历史表

```sql
CREATE TABLE approval_histories (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  
  -- 关联信息
  approval_type TEXT NOT NULL,        -- 审批类型（表名）
  approval_id INTEGER NOT NULL,       -- 审批记录ID
  
  -- 审批人信息
  approver_id INTEGER NOT NULL,       -- 审批人ID
  approver_name TEXT,                 -- 审批人姓名
  approver_role TEXT,                 -- 审批人角色
  
  -- 审批动作
  action TEXT NOT NULL,               -- approved, rejected
  comment TEXT,                       -- 审批意见
  
  -- 审批步骤
  step INTEGER,                       -- 第几步审批
  
  -- 时间戳
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  
  FOREIGN KEY (approver_id) REFERENCES users(id)
);
```

## 6. 审批流程实现

### 6.1 提交审批

```javascript
// 提交审批
async function submitApproval(type, businessId, userId) {
  const db = require('better-sqlite3')('./database.db');
  
  // 1. 获取审批流程配置
  const flow = APPROVAL_FLOWS[type];
  if (!flow) {
    throw new Error('未配置审批流程');
  }
  
  // 2. 创建审批记录
  const approvalNo = generateApprovalNo(type);
  const firstApprover = flow.steps[0].role;
  
  const result = db.prepare(`
    INSERT INTO ${type}_approvals 
    (xxx_id, approval_no, approval_step, current_approver, status, submitter_id, submitted_at)
    VALUES (?, ?, 1, ?, 'pending', ?, datetime('now'))
  `).run(businessId, approvalNo, firstApprover, userId);
  
  // 3. 更新业务状态
  db.prepare(`
    UPDATE ${type} SET status = 'pending' WHERE id = ?
  `).run(businessId);
  
  return {
    approvalId: result.lastInsertRowid,
    approvalNo,
    currentApprover: firstApprover
  };
}
```

### 6.2 审批通过

```javascript
// 审批通过
async function approveApproval(type, approvalId, userId, comment) {
  const db = require('better-sqlite3')('./database.db');
  
  // 1. 获取审批记录
  const approval = db.prepare(`
    SELECT * FROM ${type}_approvals WHERE id = ?
  `).get(approvalId);
  
  // 2. 检查审批权限
  const userRoles = getUserRoles(userId);
  if (!userRoles.includes(approval.current_approver)) {
    throw new Error('无审批权限');
  }
  
  // 3. 记录审批历史
  db.prepare(`
    INSERT INTO approval_histories 
    (approval_type, approval_id, approver_id, action, comment, step)
    VALUES (?, ?, ?, 'approved', ?, ?)
  `).run(type, approvalId, userId, comment, approval.approval_step);
  
  // 4. 获取审批流程配置
  const flow = APPROVAL_FLOWS[type];
  
  // 5. 判断是否最后一步
  if (approval.approval_step >= flow.steps.length) {
    // 审批完成
    db.prepare(`
      UPDATE ${type}_approvals 
      SET status = 'approved', 
          approval_comment = ?,
          current_approver = NULL,
          updated_at = datetime('now')
      WHERE id = ?
    `).run(comment, approvalId);
    
    // 更新业务状态
    db.prepare(`
      UPDATE ${type} SET status = 'approved' WHERE id = ?
    `).run(approval.xxx_id);
    
  } else {
    // 流转到下一步
    const nextApprover = flow.steps[approval.approval_step].role;
    
    db.prepare(`
      UPDATE ${type}_approvals 
      SET approval_step = approval_step + 1,
          current_approver = ?,
          current_approver_id = NULL,
          updated_at = datetime('now')
      WHERE id = ?
    `).run(nextApprover, approvalId);
  }
  
  return { success: true };
}
```

## 7. 待审批列表查询

### 7.1 查询逻辑

```javascript
// 获取待审批列表
async function getPendingApprovals(userId, { page = 1, pageSize = 20 }) {
  const db = require('better-sqlite3')('./database.db');
  
  // 1. 获取用户角色
  const userRoles = getUserRoles(userId);
  
  // 2. 查询各类型待审批
  const allPending = [];
  
  for (const type of APPROVAL_TYPES) {
    const pending = db.prepare(`
      SELECT 
        a.*,
        p.name as project_name,
        u.real_name as submitter_name
      FROM ${type}_approvals a
      LEFT JOIN projects p ON a.project_id = p.id
      LEFT JOIN users u ON a.submitter_id = u.id
      WHERE a.status = 'pending'
        AND a.current_approver IN (${userRoles.map(() => '?').join(',')})
    `).all(...userRoles);
    
    allPending.push(...pending.map(item => ({
      ...item,
      approval_source: type,
      source_name: APPROVAL_TYPE_NAMES[type]
    })));
  }
  
  // 3. 排序
  allPending.sort((a, b) => 
    new Date(b.created_at) - new Date(a.created_at)
  );
  
  // 4. 分页
  const total = allPending.length;
  const start = (page - 1) * pageSize;
  const list = allPending.slice(start, start + pageSize);
  
  return { list, total, page, pageSize };
}
```

## 10. 审批通知

### 10.1 待审批通知

当有新的审批任务时，系统会发送通知：

```javascript
// 发送审批通知
async function notifyApprover(approval, approverRole) {
  // 获取该角色的所有用户
  const users = db.prepare(`
    SELECT u.id, u.username, u.phone, u.email
    FROM users u
    JOIN user_roles ur ON u.id = ur.user_id
    JOIN roles r ON ur.role_id = r.id
    WHERE r.code = ? AND u.status = 'active'
  `).all(approverRole);
  
  // 发送通知
  for (const user of users) {
    await sendNotification({
      userId: user.id,
      title: '新的审批任务',
      content: `您有一个${APPROVAL_TYPE_NAMES[approval.type]}需要审批`,
      type: 'approval',
      data: {
        approvalId: approval.id,
        approvalType: approval.type
      }
    });
  }
}
```

### 10.2 审批结果通知

```javascript
// 发送审批结果通知
async function notifyResult(approval, result) {
  // 通知提交人
  await sendNotification({
    userId: approval.submitter_id,
    title: result === 'approved' ? '审批通过' : '审批被拒绝',
    content: result === 'approved'

      : `您的${APPROVAL_TYPE_NAMES[approval.type]}被拒绝：${approval.rejected_reason}`,
    type: 'approval_result'
  });
}
```

## 11. 审批统计

### 11.1 审批效率统计

```sql
-- 审批平均时长
SELECT 
  approval_type,
  AVG(julianday(updated_at) - julianday(submitted_at)) * 24 * 60 as avg_minutes
FROM xxx_approvals
WHERE status IN ('approved', 'rejected')
GROUP BY approval_type;

-- 各状态数量
SELECT 
  approval_type,
  status,
  COUNT(*) as count
FROM xxx_approvals
GROUP BY approval_type, status;
```

### 11.2 审批人效率统计

```sql
-- 审批人处理数量
SELECT 
  u.real_name,
  COUNT(*) as approval_count,
  AVG(julianday(h.created_at) - julianday(a.submitted_at)) * 24 * 60 as avg_response_minutes
FROM approval_histories h
JOIN users u ON h.approver_id = u.id
JOIN xxx_approvals a ON h.approval_id = a.id
WHERE h.action IN ('approved', 'rejected')
GROUP BY u.id
ORDER BY approval_count DESC;
```

## 12. 审批流程优化建议

### 12.1 超时提醒

```javascript
// 检查超时审批
async function checkTimeoutApprovals() {
  const db = require('better-sqlite3')('./database.db');
  
  for (const type of APPROVAL_TYPES) {
    const flow = APPROVAL_FLOWS[type];
    
    const timeoutApprovals = db.prepare(`
      SELECT * FROM ${type}_approvals
      WHERE status = 'pending'
        AND julianday('now') - julianday(submitted_at) > ?
    `).all(flow.timeout / (24 * 60 * 60 * 1000));
    
    // 发送超时提醒
    for (const approval of timeoutApprovals) {
      await notifyApprover(approval, approval.current_approver, true);
    }
  }
}
```

### 12.2 审批委托

当审批人不在时，可以委托他人审批：

```javascript
// 设置审批委托
async function setApprovalDelegation(fromUserId, toUserId, startDate, endDate) {
  db.prepare(`
    INSERT INTO approval_delegations 
    (from_user_id, to_user_id, start_date, end_date)
    VALUES (?, ?, ?, ?)
  `).run(fromUserId, toUserId, startDate, endDate);

// 检查委托
async function checkDelegation(userId) {
  const delegation = db.prepare(`
    SELECT * FROM approval_delegations
    WHERE from_user_id = ?
      AND date('now') BETWEEN start_date AND end_date
  `).get(userId);
  
  return delegation ? delegation.to_user_id : userId;
}
```

## 14. 最佳实践

1. **及时审批**: 审批人应及时处理待审批任务
2. **详细意见**: 审批时填写详细的审批意见
3. **拒绝说明**: 拒绝时必须说明原因
4. **流程监控**: 管理员应定期检查审批效率
5. **权限管理**: 合理配置审批权限，避免权限过度集中
